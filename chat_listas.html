#>>>>>>>>>>>>>>>>>>>>EJEMPLO DE CHAT BOT CON LISTAS BASICO<<<<<<<<<<<<<<<<<<<<<<<
import random
from textblob import TextBlob
from googletrans import Translator

#Lista de saludos y respuestas
saludos=["hola","que tal","buenos dias","buenas tardes","buen dia"]
respuesta_saludos=["hola, como estas",
                   "¿Como puedo ayudarte?",
                   "hola, como te va",
                   "Que hay",
                   "hola, mi rey"]

#Lista de despedidas y respuestas
despedida=["adios","bye","gracias","nos vemos","hasta luego","salir"]
respuesta_despedida=["adios, hasta pronto",
                     "fue un placer ayudarte",
                     "cuidate",
                     "nos vemos luego",
                     "bye"]

pregunta1="que haces"
pregunta2="te gusta hablar conmigo"
pregunta3=["puedes contarme un chiste",
           "cuentame un chiste",
           "puedes contarme otro chiste"]

respuesta1=["Aqui contigo,listo para hablar",
            "esperando que desees algo",
            "en este momento listo para hablar contigo"]
respuesta2=["Claro que si",
            "Me gusta hablar contigoo¡",
            "es interesante hablar contigo"]
respuesta3=["Claro¡¡¡¡¡ Porque la gallina cruzo la calle?\n Para llegar al otro lado JAJAJAJJA",
            "Claroo¡¡¡ Por que la ballena no subio a la combi?\n porque iba llena AKJKAJKJAKJA",
            "Claro, Como haces que un pan hable?\n mojandolo (pan blando)"]

pregunta_nombre=["como te llamas",
                 "cual es tu nombre",
                 "quien eres"]
respuesta_nombre=["Me llamo chato tu chatbot",
                  "Soy chato el mejor chatbot",
                  "Me conocen como chato",
                  "Soy chato tu chat bot favorito"]

#Mensaje de bienvenida del chat
print("Hola soy chato, tu chatbot de confianza, si deseas terminar escibre 'salir'")

while True:
    usuario=input("Tu: ").lower().strip()

    if usuario in saludos:
        print("Chatbot: ",random.choice(respuesta_saludos),"\n Como estas hoy?")
        translator = Translator()
        text = input("Tu: ")
        traduccion = translator.translate(text, dest='en').text

        analizar = TextBlob(traduccion)
        sentimiento = analizar.sentiment  # Analizar el sentimiento con sentiment

        print("Texto: ", text)

        if sentimiento.polarity == 0:
            print("No encuentro una emocion en tu texto")
        elif sentimiento.polarity >= -0.5 and sentimiento.polarity < 0:
            print("No te sientas asi, si lo necesitas puedes hablar con alguien")
        elif sentimiento.polarity < -0.5:
            print("No tienes porque sentirte asi, puedes buscar ayuda en fuentes confiables")
        elif sentimiento.polarity > 0 and sentimiento.polarity <= 0.5:
            print("Me alegra que estes bien")
        elif sentimiento.polarity > 0.5:
            print("Estupendo¡¡¡¡¡¡¡¡¡")
    elif usuario in pregunta1:
        print("chatbot: ",random.choice(respuesta1))
    elif usuario in pregunta2:
        print("chatbot: ",random.choice(respuesta2))
    elif usuario in pregunta3:
        print("chatbot: ",random.choice(respuesta3))
    elif usuario in pregunta_nombre:
        print("chatbot: ",random.choice(respuesta_nombre))
    elif usuario in despedida:
        print("Chatbot: ",random.choice(respuesta_despedida))
        break

Glosario de Funciones

googletrans
Es una biblioteca de Python que permite la traducción de textos mediante la API de Google Translate
translator
Es el nombre de la clase utilizada googletranspara realizar la traducción
While true
Es una estructura de control en Python que crea un bucle que se ejecuta indefinidamente, a menos que se use la instrucción breakpara salir de él.
if
Inicia una sentencia condicional que ejecuta un bloque de código solo si una condición es verdadera
Textblob
TextBlob es una biblioteca de Python para el procesamiento de datos textuales. Proporciona una API sencilla para adentrarse en tareas comunes de procesamiento del lenguaje natural
elif
Es una forma común de controlar el flujo de un programa , lo que te permite ejecutar bloques de código específicos.
